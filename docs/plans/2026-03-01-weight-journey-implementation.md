**Step 3: Create the data access module**

Create `src/lib/entries.ts` — server-only module for reading/writing the JSON file:

```ts
import fs from "fs";
import path from "path";
import { Entry } from "./types";

const DATA_PATH = path.join(process.cwd(), "data", "entries.json");

export function readEntries(): Entry[] {
  const raw = fs.readFileSync(DATA_PATH, "utf-8");
  return JSON.parse(raw);
}

export function addEntry(entry: Entry): Entry[] {
  const entries = readEntries();
  entries.push(entry);
  entries.sort((a, b) => a.date.localeCompare(b.date));
  fs.writeFileSync(DATA_PATH, JSON.stringify(entries, null, 2));
  return entries;
}
```

### Task 4: Create API Routes

**Files:**
- Create: `src/app/api/entries/route.ts`

**Step 1: Create the API route handler**

Create `src/app/api/entries/route.ts`:

```ts
import { NextResponse } from "next/server";
import { readEntries, addEntry } from "@/lib/entries";
import { Entry } from "@/lib/types";

export async function GET() {
  const entries = readEntries();
  return NextResponse.json(entries);
}

export async function POST(request: Request) {
  const body: Entry = await request.json();

  if (!body.date || body.weight == null || body.bmi == null || body.waist == null) {
    return NextResponse.json({ error: "Missing required fields" }, { status: 400 });
  }

  const entries = addEntry(body);
  return NextResponse.json(entries);
}
```

```tsx
import {
  LineChart,
  Line,
  XAxis,
  YAxis,
  CartesianGrid,
  Tooltip,
  ResponsiveContainer,
} from "recharts";
import { Entry } from "@/lib/types";

type Metric = "weight" | "bmi" | "waist";

const METRIC_CONFIG: Record<Metric, { label: string; unit: string; color: string }> = {
  weight: { label: "Weight", unit: "lbs", color: "#2563eb" },
  bmi: { label: "BMI", unit: "", color: "#16a34a" },
  waist: { label: "Waist", unit: "in", color: "#dc2626" },
};

interface MetricChartProps {
  data: Entry[];
  metric: Metric;
}

export function MetricChart({ data, metric }: MetricChartProps) {
  const config = METRIC_CONFIG[metric];

  return (
    <ResponsiveContainer width="100%" height="100%">
      <LineChart data={data} margin={{ top: 20, right: 30, left: 20, bottom: 20 }}>
        <CartesianGrid strokeDasharray="3 3" />
        <XAxis dataKey="date" />
        <YAxis
          domain={["auto", "auto"]}
          label={{ value: config.unit, angle: -90, position: "insideLeft" }}
        />
        <Tooltip />
        <Line
          type="monotone"
          dataKey={metric}
          stroke={config.color}
          strokeWidth={2}
          dot={{ r: 4 }}
          activeDot={{ r: 6 }}
        />
      </LineChart>
    </ResponsiveContainer>
  );
}

export type { Metric };
export { METRIC_CONFIG };
```

```tsx
"use client";

import { useState } from "react";
import { Button } from "@/components/ui/button";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogTrigger,
} from "@/components/ui/dialog";
import { Input } from "@/components/ui/input";
import { Label } from "@/components/ui/label";
import { Entry } from "@/lib/types";

interface AddEntryDialogProps {
  onAdd: (entry: Entry) => void;
}

export function AddEntryDialog({ onAdd }: AddEntryDialogProps) {
  const [open, setOpen] = useState(false);
  const [date, setDate] = useState(new Date().toISOString().split("T")[0]);
  const [weight, setWeight] = useState("");
  const [bmi, setBmi] = useState("");
  const [waist, setWaist] = useState("");

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault();

    onAdd({
      date,
      weight: parseFloat(weight),
      bmi: parseFloat(bmi),
      waist: parseFloat(waist),
    });

    setWeight("");
    setBmi("");
    setWaist("");
    setOpen(false);
  };

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <DialogTrigger asChild>
        <Button size="lg">Add Entry</Button>
      </DialogTrigger>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Add New Entry</DialogTitle>
        </DialogHeader>
        <form onSubmit={handleSubmit} className="flex flex-col gap-4">
          <div className="flex flex-col gap-2">
            <Label htmlFor="date">Date</Label>
            <Input
              id="date"
              type="date"
              value={date}
              onChange={(e) => setDate(e.target.value)}
              required
            />
          </div>
          <div className="flex flex-col gap-2">
            <Label htmlFor="weight">Weight (lbs)</Label>
            <Input
              id="weight"
```

### Task 8: Wire Everything Together on the Main Page

**Files:**
- Modify: `src/app/page.tsx`

**Step 1: Build the main page**

Replace `src/app/page.tsx` with the full page that fetches data and renders all components:

```tsx
"use client";

import { useEffect, useState, useCallback } from "react";
import { MetricChart, Metric } from "@/components/metric-chart";
import { MetricSelector } from "@/components/metric-selector";
import { AddEntryDialog } from "@/components/add-entry-dialog";
import { Entry } from "@/lib/types";

export default function Home() {
  const [entries, setEntries] = useState<Entry[]>([]);
  const [metric, setMetric] = useState<Metric>("weight");

  const fetchEntries = useCallback(async () => {
    const res = await fetch("/api/entries");
    const data = await res.json();
    setEntries(data);
  }, []);

  useEffect(() => {
    fetchEntries();
  }, [fetchEntries]);

  const handleAdd = async (entry: Entry) => {
    const res = await fetch("/api/entries", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify(entry),
    });
    const data = await res.json();
    setEntries(data);
  };

  return (
    <main className="flex flex-col min-h-screen p-6 gap-4">
      <MetricSelector selected={metric} onSelect={setMetric} />
      <div className="flex-1 min-h-0">
        <MetricChart data={entries} metric={metric} />
      </div>
      <div className="flex justify-center">
        <AddEntryDialog onAdd={handleAdd} />
      </div>
    </main>
  );
}
```
